Schema Definitions

DeviceInfo: object

uuid: string
uptime: integer (int64)

device uptime in unix time

resolution: string

camera resolution. Its format is "widthxheight"

video_codec: string , x ∈ { h.264 , h.265 }

video codec

Example
{
  "uuid": "123e4567-e89b-12d3-a456-426614174000",
  "uptime": "integer (int64)",
  "resolution": "1024x768",
  "video_codec": "h.265"
}

SDP: object

data: string
valid_until: string (date)

data is valid until this specified date in UTC.

Example
{
  "data": "",
  "valid_until": "2022-02-02T00:00:00.000Z"
}

SystemInfo: object

firmwareVersion: string
stun_servers: string[]
sdcard_free: integer

free sd card disk in GB

sdcard_total: integer

total sd card disk in GB

Example
{
  "firmwareVersion": "1.0.1",
  "stun_servers": [
    "stun.l.google.com:19302"
  ],
  "sdcard_free": 10,
  "sdcard_total": 10
}

User: object

id: integer (int64)
firstName: string
lastName: string
email: string
password: string
phone: string
Example
{
  "id": 10,
  "firstName": "John",
  "lastName": "James",
  "email": "john@email.com",
  "password": "12345",
  "phone": "12345"
}
